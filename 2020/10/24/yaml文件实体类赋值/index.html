package cn.tj.springboot.entity;

import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
@ConfigurationProperties(prefix = "person")
public class Person &#123;
    private Integer id;
    private String name;
    private List<String> hobby;

    public Integer getId() &#123;
        return id;
    &#125;

    public void setId(Integer id) &#123;
        this.id = id;
    &#125;

    public String getName() &#123;
        return name;
    &#125;

    public void setName(String name) &#123;
        this.name = name;
    &#125;

    public List<String> getHobby() &#123;
        return hobby;

通过@ConfigurationProperties(prefix = “person”)获取实体类属性的值 ，只有组件是容器中的组件，才能使用容器提供的@ConfigurationProperties功能
